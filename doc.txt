    <script>
        const questions = [
            {
                question: "What is the name of the protagonist in 'Naruto'?",
                options: ["Naruto Uzumaki", "Sasuke Uchiha", "Kakashi Hatake", "Sakura Haruno"],
                answer: "Naruto Uzumaki"
            },
            {
                question: "Which anime features a character named Luffy searching for the One Piece treasure?",
                options: ["Bleach", "One Piece", "Dragon Ball", "Attack on Titan"],
                answer: "One Piece"
            },
            {
                question: "In 'Attack on Titan', what are the giant humanoid creatures that threaten humanity called?",
                options: ["Titans", "Kaiju", "Colossus", "Oni"],
                answer: "Titans"
            }
        ];

        let currentQuestion = 0;
        let score = 0;
        let startTime;
        let timerInterval;

        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
        }

        function startQuiz() {
            shuffleArray(questions);
            currentQuestion = 0;
            score = 0;
            startTime = Date.now();

            const answer = confirm("Are you ready for an anime trivia challenge?");
            if (answer) {
                document.getElementById('quiz-container').style.display = "block";
                showQuestion();
                startTimer();
            } else {
                alert("No worries! Enjoy the rest of the page.");
            }
        }

        function startTimer() {
            timerInterval = setInterval(updateTimer, 1000);
        }

        function updateTimer() {
            const elapsedTime = Math.floor((Date.now() - startTime) / 1000);
            document.getElementById('timer').textContent = `Time: ${elapsedTime}s`;
        }

        function stopTimer() {
            clearInterval(timerInterval);
        }

        function showQuestion() {
            const questionElement = document.getElementById('question');
            const optionsElement = document.getElementById('options');
            const resultElement = document.getElementById('result');

            questionElement.textContent = questions[currentQuestion].question;
            optionsElement.innerHTML = '';

            questions[currentQuestion].options.forEach(option => {
                const button = document.createElement('button');
                button.textContent = option;
                button.classList.add('option');
                button.onclick = () => checkAnswer(option);
                optionsElement.appendChild(button);
            });

            resultElement.textContent = `Score: ${score}/${currentQuestion}`;
        }

        function checkAnswer(selectedAnswer) {
            const elapsedTime = Math.floor((Date.now() - startTime) / 1000);
            if (selectedAnswer === questions[currentQuestion].answer) {
                score++;
            }
            currentQuestion++;
            if (currentQuestion < questions.length) {
                showQuestion();
            } else {
                showResult(elapsedTime);
            }
        }

        function showResult(elapsedTime) {
            stopTimer();

            const resultElement = document.getElementById('result');
            resultElement.textContent = `Final Score: ${score}/${questions.length}`;

            const scoreboardElement = document.getElementById('scoreboard');
            scoreboardElement.innerHTML = '';
            const heading = document.createElement('h2');
            heading.textContent = 'Leaderboard';
            scoreboardElement.appendChild(heading);

            const listItem = document.createElement('li');
            listItem.textContent = `Score: ${score}/${questions.length} - Time: ${elapsedTime}s`;
            scoreboardElement.appendChild(listItem);

            // Show the About Me section
            document.getElementById('about-me-container').style.display = "block";
        }
    </script>
